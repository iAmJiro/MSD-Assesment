using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Review
{
    public class People
    {
        /*
        1: Readonly runs only once but we are talking about a date, a year from now, this would be out of date
        2: The constructor defaults to age 15. In a BirthingUnit, this should default to age 0
        2 continued: A fix would be Utc.Now and leave it at that if dob has no value
        */
        private static readonly DateTimeOffset Under16 = DateTimeOffset.UtcNow.AddYears(-15);

        /*
        Not too smelly, it's just that a person would most likely have a "firstName" and "lastName" seperately
        Not just bundled into the "Name"
        "null" could also be passed here and that might cause issues, the constructor must ensure it isn't "null"
        */
        public string Name { get; private set; }

        /*
        Initially "DateTimeOffset"
        Later on "DOB" is passed on to the People parameter as "DateTime"
        This would force the compiler to do an implicit conversion for which can create innaccuracy
        */
        public DateTimeOffset DOB { get; private set; }

        /*
        1. This chaining loses the "Time" part of "DateTime", it loses precision
        2. This overrides the what the correct dob should be into a date a long time ago
        */
        public People(string name) : this(name, Under16.Date) { }

        /*
        This is where it was passed on as "DateTime" rather than how it was originally declared
        Other Universities might have been taught this way, but I was taught to write it as this.name = name for safety reasons
        We don't want to accidentaly write it as "name" rather than "Name"
        However this is purely personal choice, this doesn't make the code smelly, rather I just want to point it out
        */
        public People(string name, DateTime dob)
        {
            Name = name;
            DOB = dob;
        }

        /*
        The smelliest from this class for me is 
        private static readonly DateTimeOffset Under16 = DateTimeOffset.UtcNow.AddYears(-15);
        Why 15? a new born should be 0
        Question for the original author for clarification
        "Is this "People" class one of many class for which it's civilization has to start at 15?
        Additionally, why is it named "Under16" then have "-15" as an argument, shouldn't it be -16 or 16? the name adds more to the confusion
        */
    }

    public class BirthingUnit
    {
        /// <summary>
        /// MaxItemsToRetrieve
        /// </summary>
        private List<People> _people; 

        public BirthingUnit() {
            _people = new List<People>();
        }

        /// <summary>
        /// GetPeoples
        /// </summary>
        /// <param name="j"></param>
        /// <returns>List<object></returns>
        public List<People> GetPeople(int i)
        {
            for (int j = 0; j < i; j++) {
                try
                {
                    // Creates a dandon Name
                    string name = string.Empty;
                    var random = new Random();
                    if (random.Next(0, 1) == 0) {
                        name = "Bob";
                    }
                    else { name = "Betty"; }
                    // Adds new people to the list
                    _people.Add(new People(name, DateTime.UtcNow.Subtract(new TimeSpan(random.Next(18, 85) * 356, 0, 0, 0))));
                }
                catch (Exception e)
                {
                    // Dont think this should ever happen
                    throw new Exception("Something failed in user creation");
                }
            }
            return _people;
        }

        private IEnumerable<People> GetBobs(bool olderThan30) {
            return olderThan30 ? _people.Where(x => x.Name == "Bob" && x.DOB >= DateTime.Now.Subtract(new TimeSpan(30 * 356, 0, 0, 0))) : _people.Where(x => x.Name == "Bob");
        }

        public string GetMarried(People p, string lastName)
        {
            if (lastName.Contains("test"))
                return p.Name;
            if ((p.Name.Length + lastName).Length > 255) {
                (p.Name + " " + lastName).Substring(0, 255);
            }

            return p.Name + " " + lastName;
        }
    }
